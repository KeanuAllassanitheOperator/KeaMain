php.ini

; Set the maximum time in seconds a script is allowed to parse input data, like POST and GET
max_input_time = 300

max_input_vars = 3000

; Set the maximum amount of memory a script may consume
memory_limit = 512M

; Define the maximum size of POST data that PHP will accept

; Specify the maximum size of an uploaded file

opcache.validate_timestamps = 1

opcache.revalidate_freq = 2
